```javascript
for (let i = 1; i <= N; i++) {
  arr[i] = arr[i - 1] + i;
}

const result = Array(N).fill(0);
for (let i = 0; i < K; i++) {
  let cnt = 0;
  let result = 0;
  for (let j = 0; j < questions[i]; j++) {
    if (students[j] !== Q) cnt++; //일치 하지 않을때 cnt올리기
    else {
      //일치하면 지금까지 cnt기준으로 result에 더하고 0 초기화
      result += arr[cnt];
      cnt = 0;
    }
  }
  if (cnt !== 0) {
    result += arr[cnt];
  }
  console.log(result);
}
```
